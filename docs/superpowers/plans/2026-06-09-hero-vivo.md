# Hero più vivo — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.
>
> **NB:** questa feature è pianificata ma **non ancora schedulata per l'implementazione** — verrà eseguita in blocco con le altre feature progettate. Quando si esegue, partire da qui.

**Goal:** Rendere l'Hero una dimostrazione viva della metrica taxi: cursore che illumina un rombo (taxi-circonferenza) + animazione "taxi vs diagonale" con estremi trascinabili.

**Architecture:** Estensione dell'IIFE `Hero` in `index.html`. Nuovo canvas `#hero-taxi` sopra le griglie e sotto il contenuto. Il cursore passa da maschera CSS circolare a un rivelo a rombo disegnato su canvas. La geometria riusa helper già testati di `ConicMath` (`taxiDist`, `l1Corner`, `pointAtArcFraction`); nessuna nuova funzione pura. Verifica: `node --check` dell'intera pagina + controllo visivo (è codice di rendering).

**Tech Stack:** Vanilla JS/Canvas inline in `index.html`. Pointer events (mouse+touch). `prefers-reduced-motion` rispettato.

**Spec:** `docs/superpowers/specs/2026-06-09-hero-vivo-design.md`

**Vincoli:** tutto inline; nessuna libreria; titolo/sottotitolo/bottone invariati; mantenere griglia che scorre + blob; `ConicMath` non si tocca.

## Task 2: Riscrivere l'IIFE `Hero` (cursore-rombo + taxi vs diagonale + drag + lifecycle)

**Files:**
- Modify: `index.html` — l'IIFE `const Hero = (() => { ... })();`

Questo task riscrive l'intero IIFE `Hero` (unità coesa di rendering/interazione). Verifica con `node --check` + checklist visiva.

- [ ] **Step 1: Sostituire l'IIFE `Hero`**

Trovare l'intero blocco da `const Hero = (() => {` fino a `})();` (la sua chiusura, subito prima di `// ── GraphCalc ──`) e sostituirlo INTERAMENTE con:
```js
const Hero = (() => {
  let rafId = null;
  let offsetX = 0, offsetY = 0;
  let mouseX = -9999, mouseY = -9999;
  let bgCanvas, bgCtx, cursorCanvas, cursorCtx, taxiCanvas, taxiCtx;
  let pulsePhase = 0, taxiPhase = 0.12;
  let A = null, B = null, dragging = null, placed = false, visible = false;
  const reduced = !!(window.matchMedia && window.matchMedia('(prefers-reduced-motion: reduce)').matches);

  const CELL = 40;
  const CURSOR_R0 = 150, CURSOR_AMP = 18;

  function _snap(v) { return Math.round(v / CELL) * CELL; }

  function _defaultAB() {
    const w = taxiCanvas.width, h = taxiCanvas.height;
    const cx = _snap(w / 2), baseY = _snap(h * 0.72);
    A = { x: cx - 3 * CELL, y: baseY };
    B = { x: cx + 2 * CELL, y: baseY - 3 * CELL };
  }

  function resizeCanvases() {
    const w = window.innerWidth, h = window.innerHeight;
    [bgCanvas, cursorCanvas, taxiCanvas].forEach(c => { c.width = w; c.height = h; });
    if (!placed) _defaultAB();
    frame();
  }

  function drawGrid(ctx, w, h, ox, oy) {
    ctx.clearRect(0, 0, w, h);
    ctx.strokeStyle = 'white';
    ctx.lineWidth = 1;
    const startX = ((ox % CELL) + CELL) % CELL;
    for (let x = startX - CELL; x < w + CELL; x += CELL) { ctx.beginPath(); ctx.moveTo(x, 0); ctx.lineTo(x, h); ctx.stroke(); }
    const startY = ((oy % CELL) + CELL) % CELL;
    for (let y = startY - CELL; y < h + CELL; y += CELL) { ctx.beginPath(); ctx.moveTo(0, y); ctx.lineTo(w, y); ctx.stroke(); }
  }

  // ── Cursore: rivela un ROMBO (taxi-circonferenza) attorno al mouse ──
  function drawCursorRombo() {
    const w = cursorCanvas.width, h = cursorCanvas.height;
    if (mouseX < 0) { cursorCtx.clearRect(0, 0, w, h); return; }
    drawGrid(cursorCtx, w, h, offsetX, offsetY);
    const R = CURSOR_R0 + CURSOR_AMP * Math.sin(pulsePhase);
    function rombo(ctx) {
      ctx.beginPath();
      ctx.moveTo(mouseX, mouseY - R); ctx.lineTo(mouseX + R, mouseY);
      ctx.lineTo(mouseX, mouseY + R); ctx.lineTo(mouseX - R, mouseY);
      ctx.closePath();
    }
    cursorCtx.save();
    cursorCtx.globalCompositeOperation = 'destination-in';
    const g = cursorCtx.createRadialGradient(mouseX, mouseY, 0, mouseX, mouseY, R);
    g.addColorStop(0, 'rgba(0,0,0,1)'); g.addColorStop(1, 'rgba(0,0,0,0.12)');
    cursorCtx.fillStyle = g; rombo(cursorCtx); cursorCtx.fill();
    cursorCtx.restore();
    cursorCtx.save();
    cursorCtx.strokeStyle = 'rgba(245,158,11,0.45)'; cursorCtx.lineWidth = 1;
    rombo(cursorCtx); cursorCtx.stroke();
    cursorCtx.restore();
  }

  // ── Taxi vs diagonale ──
  function _label(ctx, text, x, y, color) {
    ctx.save();
    ctx.font = '12px "IBM Plex Mono", monospace';
    ctx.fillStyle = color; ctx.strokeStyle = 'rgba(0,0,0,0.65)'; ctx.lineWidth = 3;
    ctx.textAlign = 'center';
    ctx.strokeText(text, x, y); ctx.fillText(text, x, y);
    ctx.restore();
  }
  function _handle(ctx, p) {
    ctx.save();
    ctx.fillStyle = 'rgba(255,255,255,0.9)';
    ctx.beginPath(); ctx.arc(p.x, p.y, 5, 0, Math.PI * 2); ctx.fill();
    ctx.restore();
  }
  // p in [0,1) -> frazione d'arco [0,1] con ping-pong e pause agli estremi
  function _travel(p) {
    const hd = 0.1;
    let q = p < 0.5 ? p / 0.5 : (1 - p) / 0.5; // triangolo 0..1..0
    q = (q - hd) / (1 - 2 * hd);
    return Math.max(0, Math.min(1, q));
  }
  function drawTaxi() {
    const w = taxiCanvas.width, h = taxiCanvas.height;
    taxiCtx.clearRect(0, 0, w, h);
    if (!A || !B) return;
    const corner = ConicMath.l1Corner(A, B); // {x:B.x, y:A.y}
    // diagonale euclidea (fantasma)
    taxiCtx.save();
    taxiCtx.strokeStyle = 'rgba(255,255,255,0.22)'; taxiCtx.lineWidth = 1.5; taxiCtx.setLineDash([6, 6]);
    taxiCtx.beginPath(); taxiCtx.moveTo(A.x, A.y); taxiCtx.lineTo(B.x, B.y); taxiCtx.stroke();
    taxiCtx.restore();
    // percorso Manhattan (evidenziato)
    taxiCtx.save();
    taxiCtx.strokeStyle = 'rgba(245,158,11,0.9)'; taxiCtx.lineWidth = 3;
    taxiCtx.lineJoin = 'round'; taxiCtx.lineCap = 'round';
    taxiCtx.shadowColor = 'rgba(245,158,11,0.6)'; taxiCtx.shadowBlur = 10;
    taxiCtx.beginPath(); taxiCtx.moveTo(A.x, A.y); taxiCtx.lineTo(corner.x, corner.y); taxiCtx.lineTo(B.x, B.y); taxiCtx.stroke();
    taxiCtx.restore();
    // taxi (pallino) lungo la L
    const P = ConicMath.pointAtArcFraction([A, corner, B], _travel(taxiPhase), false);
    if (P) {
      taxiCtx.save();
      taxiCtx.fillStyle = '#fde68a'; taxiCtx.shadowColor = 'rgba(253,230,138,0.9)'; taxiCtx.shadowBlur = 12;
      taxiCtx.beginPath(); taxiCtx.arc(P.x, P.y, 5, 0, Math.PI * 2); taxiCtx.fill();
      taxiCtx.restore();
    }
    _handle(taxiCtx, A); _handle(taxiCtx, B);
    const d1 = ConicMath.taxiDist(A, B) / CELL;
    const d2 = Math.hypot(B.x - A.x, B.y - A.y) / CELL;
    _label(taxiCtx, `d₁ = ${d1.toFixed(0)}`, corner.x, corner.y + (B.y >= A.y ? 16 : -10), 'rgba(245,158,11,0.95)');
    _label(taxiCtx, `d₂ ≈ ${d2.toFixed(1)}`, (A.x + B.x) / 2, (A.y + B.y) / 2 - 8, 'rgba(255,255,255,0.7)');
  }

  function frame() {
    if (!reduced) {
      offsetX = (offsetX + 0.5) % CELL;
      offsetY = (offsetY + 0.5) % CELL;
      pulsePhase += 0.05;
      if (!dragging) taxiPhase = (taxiPhase + 0.0035) % 1;
    }
    drawGrid(bgCtx, bgCanvas.width, bgCanvas.height, offsetX, offsetY);
    drawCursorRombo();
    drawTaxi();
  }
  function loop() { frame(); rafId = requestAnimationFrame(loop); }

  function onMouseMove(e) { mouseX = e.clientX; mouseY = e.clientY; if (reduced && visible) frame(); }

  // ── drag di A/B ──
  function _pt(e) { const r = taxiCanvas.getBoundingClientRect(); return { x: e.clientX - r.left, y: e.clientY - r.top }; }
  function _hit(x, y) {
    if (A && Math.hypot(A.x - x, A.y - y) < 16) return 'A';
    if (B && Math.hypot(B.x - x, B.y - y) < 16) return 'B';
    return null;
  }
  function _clamp(p) {
    const w = taxiCanvas.width, h = taxiCanvas.height;
    p.x = Math.max(CELL, Math.min(w - CELL, p.x));
    p.y = Math.max(CELL, Math.min(h - CELL, p.y));
  }
  function onDown(e) {
    const { x, y } = _pt(e);
    const hit = _hit(x, y);
    if (!hit) return;                 // non-handle: lascia scrollare/click
    dragging = hit; placed = true;
    try { taxiCanvas.setPointerCapture(e.pointerId); } catch (_) {}
    taxiCanvas.style.cursor = 'grabbing';
    e.preventDefault();
    drawTaxi();
  }
  function onMove(e) {
    const { x, y } = _pt(e);
    if (dragging) {
      const p = dragging === 'A' ? A : B;
      p.x = _snap(x); p.y = _snap(y); _clamp(p);
      e.preventDefault();
      drawTaxi();
      return;
    }
    taxiCanvas.style.cursor = _hit(x, y) ? 'grab' : 'default';
  }
  function onUp(e) {
    if (!dragging) return;
    dragging = null;
    try { taxiCanvas.releasePointerCapture(e.pointerId); } catch (_) {}
    taxiCanvas.style.cursor = 'default';
  }

  function start() { if (!reduced && rafId == null) rafId = requestAnimationFrame(loop); }
  function stop()  { if (rafId != null) { cancelAnimationFrame(rafId); rafId = null; } }

  function init() {
    bgCanvas = document.getElementById('hero-grid-bg');
    cursorCanvas = document.getElementById('hero-grid-cursor');
    taxiCanvas = document.getElementById('hero-taxi');
    bgCtx = bgCanvas.getContext('2d');
    cursorCtx = cursorCanvas.getContext('2d');
    taxiCtx = taxiCanvas.getContext('2d');

    resizeCanvases();
    window.addEventListener('resize', resizeCanvases);
    document.addEventListener('mousemove', onMouseMove);
    taxiCanvas.addEventListener('pointerdown', onDown);
    taxiCanvas.addEventListener('pointermove', onMove);
    taxiCanvas.addEventListener('pointerup', onUp);
    taxiCanvas.addEventListener('pointercancel', onUp);

    const hero = document.getElementById('hero');
    const obs = new IntersectionObserver(es => {
      es.forEach(en => { visible = en.isIntersecting; if (visible) start(); else stop(); });
    }, { threshold: 0 });
    obs.observe(hero);

    frame(); // primo disegno
  }

  return { init, start, stop };
})();
```
